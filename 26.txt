{

Q23. /*
1 10 11 20 21
2 9 12 19 22
3 8 13 18 23
4 7 14 17 24
5 6 15 16 25 
*/


#include <stdio.h>

int main()
{
    int n=5;
  for(int i=1;i<=5;i++){
      
    for(int j=0;j<5;j++){
        int x;
        if(j%2==0){
            x=i+j*n;
        }
        else{
            x=(j+1)*n-(i-1);
        }
        printf("%d ",x);
    }
   
      printf("\n");
  }
    return 0;
}
